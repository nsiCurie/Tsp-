# xA,yA,xB,YB position de la plante entre les deux positions A et B
from math import sqrt,pi
def surface(xA, yA, xB, yB):
    """ calcule la surface d'un triangle OAB (O est l'origine du repre) 
    xA, yA, xB, yB sont les coordonnes des sommets A et B """
    a = sqrt(xA**2 + yA**2)
    b = sqrt(xB**2 + yB**2)
    c = sqrt((xB - xA)**2 + (yB - yA)**2)
    p = (a + b + c) / 2 
    s = sqrt(p * (p-a) * (p-b) * (p-c)) # formule de Hron
    return s
